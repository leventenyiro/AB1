select pkod, pnev
from projekt
natural join szallit
natural join cikk
where szin like 'zold'
group by pkod, pnev

    from cikk

from szallit
natural join szallito
group by szkod, sznev
having count(distinct pkod) = (
    select count(*)
    from projekt)

    and sz1.gyumolcs = 'alma'
    and sz2.gyumolcs <> 'alma'
